pk = primary key
cpk = composite primary key
fk = foreign key

wedding
	pk int id
	float budget
	date wedding_date
	date rsvp_by_date
	
asset
	pk int id
	varchar email
	varchar name
	varchar phone
	float price
	varchar address
	fk int type_id
	
asset_type
	pk int id
	varchar description (caterer, florist, photographer, musician, venue)
	
asset_booking
	cpk fk int asset_id
	cpk fk int wedding_id
	date booking_date
	
app_users
	pk varchar email
	varchar username
	varchar password
	varchar phone
	
employee
	pk fk user_email
	
betrothed
	pk fk user_email
	wedding_id
	
attendee
	pk int id
	fk user_email
	fk wedding_id
	rsvp_answer (bit, default null, 1 for yes, 0 for no)
	plus_one_rsvp (boolean, default null, 1 for yes, 0 for no)

meal_choice
	pk id
	description (meat, fish, veggie)
	
meal_order
	pk fk attendee_id
	lunch_choice
	dinner_choice
	
meal_order_plus_one
	pk fk attendee_id
	lunch_choice
	dinner_choice
